function fmt$(n){ return "$" + Math.round(n).toLocaleString(); }
function barColor(val, target){ const p=(val/target)*100; return p>=100?"#F4B942":p>=70?"#f5a623":"#ff5f57"; }

function setBar(barId, badgeId, val, target){
  const p = Math.min(100, target>0?(val/target)*100:0);
  const col = barColor(val, target);
  const bar = document.getElementById(barId);
  bar.style.width = p+"%"; bar.style.background = col;
  const badge = document.getElementById(badgeId);
  const ok = val >= target;
  badge.style.background = ok?"rgba(244,185,66,0.1)":"rgba(255,95,87,0.08)";
  badge.style.color = ok?"#F4B942":"#ff5f57";
  badge.textContent = (ok?"✓ ":"")+"target: "+target;
}

  setBar("bar-quotes","badge-quotes",q,TARGETS.quotes);

    rcl.textContent=closePct+"%"; rcl.style.color=parseInt(closePct)>=20?"#F4B942":"#ff5f57";

  } else { document.getElementById("rates").style.display="none"; }

  // Score
  const s1=Math.min(1,q/TARGETS.quotes), s2=Math.min(1,c/TARGETS.calls);

  const s5=Math.min(1,avgDeal/3500);

  if(hasData){

    else{vt="BELOW TARGET";vb="rgba(255,95,87,0.1)";vc="#ff5f57";}
    v.textContent=vt;v.style.background=vb;v.style.color=vc;v.style.border="1px solid "+vc;

    const sb=(bid,vid,val,target,disp)=>{
      const p=Math.min(100,target>0?(val/target)*100:0);
      const col=barColor(val,target);
      document.getElementById(bid).style.width=p+"%";
      document.getElementById(bid).style.background=col;
      document.getElementById(vid).textContent=disp||val;
      document.getElementById(vid).style.color=col;
    };
    sb("sb-q","sb-val-q",q,TARGETS.quotes,q+" / "+TARGETS.quotes);
    sb("sb-c","sb-val-c",c,TARGETS.calls,c+" / "+TARGETS.calls);
    sb("sb-cv","sb-val-cv",cv,TARGETS.connected,cv+" / "+TARGETS.connected);
    sb("sb-f","sb-val-f",f,TARGETS.followups,f+" / "+TARGETS.followups);
    sb("sb-d","sb-val-d",avgDeal,3500,fmt$(avgDeal)+" / "+fmt$(3500));
  } else { document.getElementById("score-card").style.display="none"; }

    document.getElementById("rev-card").style.display="flex";
    document.getElementById("rev-amount").textContent=fmt$(rev);
    document.getElementById("rev-sub").textContent=productRows.length+" container"+(productRows.length>1?"s":"")+" · avg "+fmt$(avgDeal);
    const closePct=q>0?((productRows.length/q)*100).toFixed(0):0;
    document.getElementById("rev-rate").textContent=closePct+"%";
    document.getElementById("rev-close-sub").textContent=q+" quotes → "+productRows.length+" closed";
  } else { document.getElementById("rev-card").style.display="none"; }

  // Submit button
  const btn=document.getElementById("submit-btn");
  if(hasData){ btn.classList.remove("disabled");btn.classList.add("active"); }
  else{ btn.classList.add("disabled");btn.classList.remove("active"); }
}

// ── Products ─────────────────────────────────────────────────────────────────
function buildOptions(){
  const groups={"20ft Containers":[],"40ft Containers":[]};
  PRODUCTS.forEach(p=>{ if(p.group==="20ft") groups["20ft Containers"].push(p); else groups["40ft Containers"].push(p); });
  let html='<option value="">— Select container —</option>';
  Object.entries(groups).forEach(([g,items])=>{
    html+=`<optgroup label="${g}">`;
    items.forEach(p=>html+=`<option value="${p.label}" data-price="${p.price}">${p.label}</option>`);
    html+='</optgroup>';
  });
  return html;
}

function addProduct(){
  const id="pr_"+Date.now();
  productRows.push({id,label:"",price:0});
  const wrap=document.getElementById("prod-entries");
  const div=document.createElement("div");
  div.className="prod-row"; div.id=id;
  div.innerHTML=`
    <select class="prod-select" onchange="onSelect('${id}',this)">${buildOptions()}</select>
    <input class="prod-price" type="number" placeholder="$" min="0" step="100" oninput="onPrice('${id}',this)"/>
    <button class="remove-btn" onclick="removeProduct('${id}')">✕</button>`;
  wrap.appendChild(div);
  updateUnitsBadge();
}

function onSelect(id, sel){
  const opt=sel.options[sel.selectedIndex];
  const price=parseInt(opt.getAttribute("data-price")||0);
  const row=productRows.find(r=>r.id===id);
  if(row){ row.label=sel.value; row.price=price; }
  const priceInput=document.getElementById(id).querySelector(".prod-price");
  if(price) priceInput.value=price;
  calc();
}

function onPrice(id, input){
  const row=productRows.find(r=>r.id===id);
  if(row) row.price=parseInt(input.value)||0;
  calc();
}

function removeProduct(id){
  productRows=productRows.filter(r=>r.id!==id);
  document.getElementById(id)?.remove();
  updateUnitsBadge(); calc();
}

function updateUnitsBadge(){
  const b=document.getElementById("units-badge");
  const rev=productRows.reduce((s,r)=>s+(r.price||0),0);
  if(productRows.length>0){ b.style.display="inline"; b.textContent=productRows.length+" unit"+(productRows.length>1?"s":"")+" · $"+Math.round(rev).toLocaleString(); }
  else b.style.display="none";
}

// ── Submit ────────────────────────────────────────────────────────────────────
function handleSubmit(){
  const q=parseInt(document.getElementById("quotes").value)||0;
  const c=parseInt(document.getElementById("calls").value)||0;
  const cv=parseInt(document.getElementById("connected").value)||0;
  const f=parseInt(document.getElementById("followups").value)||0;
  if(!q&&!c) return;

  const rev=productRows.reduce((s,r)=>s+(r.price||0),0);
  const avgDeal=productRows.length>0?rev/productRows.length:0;
  const s1=Math.min(1,q/5),s2=Math.min(1,c/15),s3=Math.min(1,cv/6),s4=Math.min(1,f/5),s5=Math.min(1,avgDeal/3500);
  const score=((s1+s2+s3+s4+s5)/5*5).toFixed(1);
  const prodStr=productRows.filter(r=>r.label).map(r=>r.label).join(" | ");

  // Fill hidden form fields
  document.getElementById("f-date").value=document.getElementById("date").value;
  document.getElementById("f-rep").value=currentRep;
  document.getElementById("f-quotes").value=q;
  document.getElementById("f-calls").value=c;
  document.getElementById("f-connected").value=cv;
  document.getElementById("f-followups").value=f;
  document.getElementById("f-sold").value=productRows.length;
  document.getElementById("f-products").value=prodStr;
  document.getElementById("f-revenue").value=rev;
  document.getElementById("f-score").value=score;
  document.getElementById("f-notes").value=document.getElementById("notes").value.replace(/\n/g," ");

  // Set form action and submit into hidden iframe
  document.getElementById("submit-form").action=SCRIPT_URL;
  document.getElementById("submit-form").submit();

  // Show success (we can't read the iframe response, but if no error it went through)
  setTimeout(()=>{
    document.getElementById("submit-wrap").style.display="none";
    const ss=document.getElementById("success-state");
    ss.style.display="block";
    document.getElementById("success-msg").textContent=
      "Submitted — "+currentRep+" · "+document.getElementById("date").value;
  }, 800);
}

function resetForm(){
  document.getElementById("quotes").value="";
  document.getElementById("calls").value="";
  document.getElementById("connected").value="";
  document.getElementById("followups").value="";
  document.getElementById("notes").value="";
  document.getElementById("prod-entries").innerHTML="";
  productRows=[];
  document.getElementById("success-state").style.display="none";
  document.getElementById("submit-wrap").style.display="block";
  calc();
}
